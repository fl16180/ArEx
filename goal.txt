compatible with scikit-learn

usages:

from argotools.datasets import cdc_loader, athena_loader
from argotools import DataConfig
from argotools import Argo
from sklearn.linear_model import LassoCV

cdc_national_file = './data/US_national.csv'
athena_file = './data/athenahealth.csv'

cdc_dataset = cdc_loader(cdc_national_file)
athena_dataset = athena_loader(athena_file)

config = TSConfig()
config.register_dataset(cdc_dataset, name='CDC', type='target')
config.register_dataset(athena_dataset, name='athena', type='predictor')
config.add_AR(terms=range(1, 12), dataset='CDC', var='%ILI')
config.stack(datasets=['athena', 'AR_CDC'])

base_mod = LassoCV(cv=10)
model = Argo(config=config, model=base_mod,
             pred_start="2019-02-19", pred_end="2019-03-12",
             training="rolling", roll_window=104)
model.predict()
